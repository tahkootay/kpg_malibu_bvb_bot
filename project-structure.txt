# Структура проекта volleyball_bot/
#
# config/
# ├── config.py          # Конфигурационный файл с настройками бота
# ├── messages.py        # Шаблоны сообщений
# 
# database/
# ├── __init__.py
# ├── models.py          # Модели базы данных
# ├── database.py        # Функции для работы с базой данных
#
# handlers/
# ├── __init__.py
# ├── admin_handlers.py  # Обработчики команд администратора
# ├── user_handlers.py   # Обработчики пользовательских команд
# ├── common.py         # Общие функции для обработчиков
#
# utils/
# ├── __init__.py
# ├── formatting.py      # Функции форматирования текста
# ├── validators.py      # Функции валидации
# ├── logger.py         # Настройки логирования
#
# main.py               # Основной файл бота
# requirements.txt      # Зависимости проекта
